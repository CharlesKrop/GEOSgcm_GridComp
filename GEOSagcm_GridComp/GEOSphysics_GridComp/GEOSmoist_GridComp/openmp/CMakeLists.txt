  Process_Library.F90
  GEOS_GFDL_1M_InterfaceMod.F90
  gfdl_cloud_microphys.F90)

  SRCS ${srcs}
  DEPENDENCIES MAPL GEOS_Shared GMAO_mpeu)
  # DEPENDENCIES GEOS_Shared GMAO_mpeu MAPL Chem_Shared Chem_Base esmf)
